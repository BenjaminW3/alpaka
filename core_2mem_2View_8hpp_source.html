40  //! A memory buffer view.
41  //#############################################################################
42  template<
43  typename TBuf>
44  class View
45  {
46  private:
47  using Dim = dim::DimT<TBuf>;
48  using Buf = BufT<dev::DevT<TBuf>, ElemT<TBuf>, Dim, SpaceT<TBuf>>;
49  using MemSpace = SpaceT<TBuf>;
50 
51  public:
52  //-----------------------------------------------------------------------------
53  //! Constructor.
54  //! \param buf This can be either a memory buffer base or a memory buffer view itself.
55  //-----------------------------------------------------------------------------
56  View(
57  TBuf const & buf) :
58  m_buf(getBuf(buf)),
59  m_vOffsetsElements(Vec<Dim::value>::fromOffsets(buf)),
60  m_vExtentsElements(Vec<Dim::value>::fromExtents(buf))
61  {}
62 
63  //-----------------------------------------------------------------------------
64  //! Constructor.
65  //! \param buf This can be either a memory buffer base or a memory buffer view itself.
66  //! \param offsetsElements The offsets in elements.
67  //! \param extentsElements The extents in elements.
68  //-----------------------------------------------------------------------------
69  template<
70  typename TOffsets,
71  typename TExtents>
72  View(
73  TBuf const & buf,
74  TExtents const & extentsElements,
75  TOffsets const & relativeOffsetsElements = TOffsets()) :
76  m_buf(getBuf(buf)),
77  m_vOffsetsElements(Vec<Dim::value>::fromOffsets(relativeOffsetsElements)+Vec<Dim::value>::fromOffsets(buf)),
78  m_vExtentsElements(Vec<Dim::value>::fromExtents(extentsElements))
79  {
80  static_assert(
81  std::is_same<Dim, dim::DimT<TExtents>>::value,
82  "The base buffer and the extents are required to have the same dimensionality!");
83 
84  assert(extent::getWidth(relativeOffsetsElements) <= extent::getWidth(buf));
85  assert(extent::getHeight(relativeOffsetsElements) <= extent::getHeight(buf));
86  assert(extent::getDepth(relativeOffsetsElements) <= extent::getDepth(buf));
87  assert((offset::getOffsetX(relativeOffsetsElements)+offset::getOffsetX(buf)+extent::getWidth(extentsElements)) <= extent::getWidth(buf));
88  assert((offset::getOffsetY(relativeOffsetsElements)+offset::getOffsetY(buf)+extent::getHeight(extentsElements)) <= extent::getHeight(buf));
89  assert((offset::getOffsetZ(relativeOffsetsElements)+offset::getOffsetZ(buf)+extent::getDepth(extentsElements)) <= extent::getDepth(buf));
90  }
91 
92  public:
93  Buf m_buf;
94  Vec<Dim::value> m_vOffsetsElements;
95  Vec<Dim::value> m_vExtentsElements;
96  };
97  }
