39  //! A memory buffer view.
40  //#############################################################################
41  template<
42  typename TBuf>
43  class View
44  {
45  private:
46  using Dim = dim::DimT<TBuf>;
47  using Buf = BufT<ElemT<TBuf>, Dim, SpaceT<TBuf>>;
48  using MemSpace = SpaceT<TBuf>;
49 
50  public:
51  //-----------------------------------------------------------------------------
52  //! Constructor.
53  //! \param buf This can be either a memory buffer base or a memory buffer view itself.
54  //-----------------------------------------------------------------------------
55  View(
56  TBuf const & buf) :
57  m_buf(getBuf(buf)),
58  m_vOffsetsElements(Vec<Dim::value>::fromOffsets(buf)),
59  m_vExtentsElements(Vec<Dim::value>::fromExtents(buf))
60  {}
61 
62  //-----------------------------------------------------------------------------
63  //! Constructor.
64  //! \param buf This can be either a memory buffer base or a memory buffer view itself.
65  //! \param offsetsElements The offsets in elements.
66  //! \param extentsElements The extents in elements.
67  //-----------------------------------------------------------------------------
68  template<
69  typename TOffsets,
70  typename TExtents>
71  View(
72  TBuf const & buf,
73  TExtents const & extentsElements,
74  TOffsets const & relativeOffsetsElements = TOffsets()) :
75  m_buf(getBuf(buf)),
76  m_vOffsetsElements(Vec<Dim::value>::fromOffsets(relativeOffsetsElements)+Vec<Dim::value>::fromOffsets(buf)),
77  m_vExtentsElements(Vec<Dim::value>::fromExtents(extentsElements))
78  {
79  static_assert(
80  std::is_same<Dim, dim::DimT<TExtents>>::value,
81  "The base buffer and the extents are required to have the same dimensionality!");
82 
83  assert(extent::getWidth(relativeOffsetsElements) <= extent::getWidth(buf));
84  assert(extent::getHeight(relativeOffsetsElements) <= extent::getHeight(buf));
85  assert(extent::getDepth(relativeOffsetsElements) <= extent::getDepth(buf));
86  assert((offset::getOffsetX(relativeOffsetsElements)+offset::getOffsetX(buf)+extent::getWidth(extentsElements)) <= extent::getWidth(buf));
87  assert((offset::getOffsetY(relativeOffsetsElements)+offset::getOffsetY(buf)+extent::getHeight(extentsElements)) <= extent::getHeight(buf));
88  assert((offset::getOffsetZ(relativeOffsetsElements)+offset::getOffsetZ(buf)+extent::getDepth(extentsElements)) <= extent::getDepth(buf));
89  }
90 
91  public:
92  Buf m_buf;
93  Vec<Dim::value> m_vOffsetsElements;
94  Vec<Dim::value> m_vExtentsElements;
95  };
96  }
97  }
