105  //! Executes the kernel functor.
106  //-----------------------------------------------------------------------------
107  template<
108  typename TKernelFunctor,
109  typename... TArgs>
110  ALPAKA_FCT_HOST auto operator()(
111  TKernelFunctor && kernelFunctor,
112  TArgs && ... args) const
113  -> void
114  {
115  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
116 
117  auto const vuiGridBlockExtents(
118  workdiv::getWorkDiv<Grid, Blocks>(
119  *static_cast<workdiv::BasicWorkDiv<TDim> const *>(this)));
120  auto const vuiBlockThreadExtents(
121  workdiv::getWorkDiv<Block, Threads>(
122  *static_cast<workdiv::BasicWorkDiv<TDim> const *>(this)));
123 
124  auto const uiBlockSharedExternMemSizeBytes(
125  kernel::getBlockSharedExternMemSizeBytes<
126  typename std::decay<TKernelFunctor>::type,
127  AccCpuOmp4<TDim>>(
128  vuiBlockThreadExtents,
129  std::forward<TArgs>(args)...));
130 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
131  std::cout << BOOST_CURRENT_FUNCTION
132  << " BlockSharedExternMemSizeBytes: " << uiBlockSharedExternMemSizeBytes << " B"
133  << std::endl;
134 #endif
135  // The number of blocks in the grid.
136  UInt const uiNumBlocksInGrid(vuiGridBlockExtents.prod());
137  int const iNumBlocksInGrid(static_cast<int>(uiNumBlocksInGrid));
138  // The number of threads in a block.
139  UInt const uiNumThreadsInBlock(vuiBlockThreadExtents.prod());
140  int const iNumThreadsInBlock(static_cast<int>(uiNumThreadsInBlock));
141 
142  // `When an if(scalar-expression) evaluates to false, the structured block is executed on the host.`
143  #pragma omp target if(0)
144  {
145  #pragma omp teams num_teams(iNumBlocksInGrid) thread_limit(iNumThreadsInBlock)
146  {
147 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
148  // The first team does some checks ...
149  if((::omp_get_team_num() == 0))
150  {
151  int const iNumTeams(::omp_get_num_teams());
152  // NOTE: No std::cout in omp target!
153  printf("%s omp_get_num_teams: %d\n", BOOST_CURRENT_FUNCTION, iNumTeams);
154  if(iNumTeams <= 0) // NOTE: No throw inside target region
155  {
156  throw std::runtime_error("The CPU OpenMP4 runtime did not use a valid number of teams!");
157  }
158  }
159 #endif
160  AccCpuOmp4<TDim> acc(*static_cast<workdiv::BasicWorkDiv<TDim> const *>(this));
161 
162  if(uiBlockSharedExternMemSizeBytes > 0)
163  {
164  acc.m_vuiExternalSharedMem.reset(
165  new uint8_t[uiBlockSharedExternMemSizeBytes]);
166  }
167 
168  #pragma omp distribute
169  for(UInt b = 0u; b<uiNumBlocksInGrid; ++b)
170  {
171  Vec1<> const v1iIdxGridBlock(b);
172  auto const vuiGridBlockExtents(
173  workdiv::getWorkDiv<Grid, Blocks>(
174  *static_cast<workdiv::BasicWorkDiv<TDim> const *>(this)));
175  acc.m_vuiGridBlockIdx = mapIdx<TDim::value>(
176  v1iIdxGridBlock,
177  vuiGridBlockExtents);
178 
179  // Execute the threads in parallel.
180 
181  // Force the environment to use the given number of threads.
182  ::omp_set_dynamic(0);
183 
184  // Parallel execution of the threads in a block is required because when syncBlockThreads is called all of them have to be done with their work up to this line.
185  // So we have to spawn one OS thread per thread in a block.
186  // 'omp for' is not useful because it is meant for cases where multiple iterations are executed by one thread but in our case a 1:1 mapping is required.
187  // Therefore we use 'omp parallel' with the specified number of threads in a block.
188  #pragma omp parallel num_threads(iNumThreadsInBlock)
189  {
190 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
191  // The first thread does some checks in the first block executed.
192  if((::omp_get_thread_num() == 0) && (b == 0))
193  {
194  int const iNumThreads(::omp_get_num_threads());
195  // NOTE: No std::cout in omp target!
196  printf("%s omp_get_num_threads: %d\n", BOOST_CURRENT_FUNCTION, iNumThreads);
197  if(iNumThreads != iNumThreadsInBlock)
198  {
199  throw std::runtime_error("The CPU OpenMP4 runtime did not use the number of threads that had been required!");
200  }
201  }
202 #endif
203  std::forward<TKernelFunctor>(kernelFunctor)(
204  acc,
205  std::forward<TArgs>(args)...);
206 
207  // Wait for all threads to finish before deleting the shared memory.
208  acc.syncBlockThreads();
209  }
210 
211  // After a block has been processed, the shared memory has to be deleted.
212  acc.m_vvuiSharedMem.clear();
213  }
214  // After all blocks have been processed, the external shared memory has to be deleted.
215  acc.m_vuiExternalSharedMem.reset();
216  }
217  }
218  }
