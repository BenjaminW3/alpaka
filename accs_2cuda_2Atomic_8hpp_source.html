157  //! The CUDA accelerator atomic operation functor.
158  //-----------------------------------------------------------------------------
159  template<>
160  struct AtomicOp<
161  accs::cuda::detail::AtomicCuda,
162  ops::Add,
163  double>
164  {
165  ALPAKA_FCT_ACC_CUDA_ONLY static auto atomicOp(
166  accs::cuda::detail::AtomicCuda const &,
167  double * const addr,
168  double const & value)
169  -> double
170  {
171  // Code from: http://docs.nvidia.com/cuda/cuda-c-programming-guide/#atomic-functions
172 
173  unsigned long long int * address_as_ull(reinterpret_cast<unsigned long long int *>(addr));
174  unsigned long long int old(*address_as_ull);
175  unsigned long long int assumed;
176  do
177  {
178  assumed = old;
179  old = atomicCAS(address_as_ull, assumed, __double_as_longlong(value + __longlong_as_double(assumed)));
180  // Note: uses integer comparison to avoid hang in case of NaN (since NaN != NaN)
181  }
182  while(assumed != old);
183  return __longlong_as_double(old);
184  }
185  };
