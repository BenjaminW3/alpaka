46  //! The CUDA accelerator event.
47  //#############################################################################
48  class EventCuda
49  {
50  public:
51  //-----------------------------------------------------------------------------
52  //! Constructor.
53  //-----------------------------------------------------------------------------
54  ALPAKA_FCT_HOST EventCuda(
55  DevCuda const & dev,
56  bool bBusyWait = true) :
57  m_Dev(dev),
58  m_spCudaEvent(
59  new cudaEvent_t,
60  std::bind(&EventCuda::destroyEvent, std::placeholders::_1, std::ref(m_Dev)))
61  {
62  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
63 
64  // Set the current device.
65  ALPAKA_CUDA_CHECK(cudaSetDevice(
66  m_Dev.m_iDevice));
67  // Create the event on the current device with the specified flags. Valid flags include:
68  // cudaEventDefault: Default event creation flag.
69  // cudaEventBlockingSync : Specifies that event should use blocking synchronization.A host thread that uses cudaEventSynchronize() to wait on an event created with this flag will block until the event actually completes.
70  // cudaEventDisableTiming : Specifies that the created event does not need to record timing data.Events created with this flag specified and the cudaEventBlockingSync flag not specified will provide the best performance when used with cudaStreamWaitEvent() and cudaEventQuery().
71  ALPAKA_CUDA_CHECK(cudaEventCreateWithFlags(
72  m_spCudaEvent.get(),
73  (bBusyWait ? cudaEventDefault : cudaEventBlockingSync) | cudaEventDisableTiming));
74  }
75  //-----------------------------------------------------------------------------
76  //! Copy constructor.
77  //-----------------------------------------------------------------------------
78  ALPAKA_FCT_HOST EventCuda(EventCuda const &) = default;
79 #if (!BOOST_COMP_MSVC) || (BOOST_COMP_MSVC >= BOOST_VERSION_NUMBER(14, 0, 0))
80  //-----------------------------------------------------------------------------
81  //! Move constructor.
82  //-----------------------------------------------------------------------------
83  ALPAKA_FCT_HOST EventCuda(EventCuda &&) = default;
84 #endif
85  //-----------------------------------------------------------------------------
86  //! Assignment operator.
87  //-----------------------------------------------------------------------------
88  ALPAKA_FCT_HOST auto operator=(EventCuda const &) -> EventCuda & = default;
89  //-----------------------------------------------------------------------------
90  //! Equality comparison operator.
91  //-----------------------------------------------------------------------------
92  ALPAKA_FCT_HOST auto operator==(EventCuda const & rhs) const
93  -> bool
94  {
95  return (*m_spCudaEvent.get()) == (*rhs.m_spCudaEvent.get());
96  }
97  //-----------------------------------------------------------------------------
98  //! Equality comparison operator.
99  //-----------------------------------------------------------------------------
100  ALPAKA_FCT_HOST auto operator!=(EventCuda const & rhs) const
101  -> bool
102  {
103  return !((*this) == rhs);
104  }
105  //-----------------------------------------------------------------------------
106  //! Destructor.
107  //-----------------------------------------------------------------------------
108  ALPAKA_FCT_HOST /*virtual*/ ~EventCuda() noexcept = default;
109 
110  private:
111  //-----------------------------------------------------------------------------
112  //! Destroys the shared event.
113  //-----------------------------------------------------------------------------
114  static auto destroyEvent(
115  cudaEvent_t * event,
116  DevCuda const & dev)
117  -> void
118  {
119  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
120 
121  // Set the current device. \TODO: Is setting the current device before cudaEventDestroy required?
122  ALPAKA_CUDA_CHECK(cudaSetDevice(
123  dev.m_iDevice));
124  // In case event has been recorded but has not yet been completed when cudaEventDestroy() is called, the function will return immediately
125  // and the resources associated with event will be released automatically once the device has completed event.
126  // -> No need to synchronize here.
127  ALPAKA_CUDA_CHECK(cudaEventDestroy(*event));
128  }
129 
130  public:
131  std::shared_ptr<cudaEvent_t> m_spCudaEvent;
132  DevCuda m_Dev;
133  };
134  }
135  }
