179  //#############################################################################
180  //! Subdivides the given grid thread extents into blocks restricted by:
181  //! 1. The maximum block thread extents
182  //! 2. The maximum block thread count
183  //! 3. The requirement of the block thread extents to divide the grid thread extents without remainder
184  //!
185  //! \param vuiGridThreadExtents
186  //! The full extents of threads in the grid.
187  //! \param vuiMaxBlockThreadExtents
188  //! The maximum extents of threads in a block.
189  //! \param uiMaxBlockThreadsCount
190  //! The maximum number of threads in a block.
191  //! \param bRequireBlockThreadExtentsToDivideGridThreadExtents
192  //! If this is true, the grid thread extents will be multiples of the corresponding block thread extents.
193  //! NOTE: If vuiGridThreadExtents is prime (or otherwise bad chosen) in a dimension, the block thread extent will be one in this dimension.
194  //#############################################################################
195  template<
196  typename TDim>
197  ALPAKA_FCT_HOST auto subdivideGridThreads(
198  Vec<TDim> const & vuiGridThreadExtents,
199  Vec<TDim> const & vuiMaxBlockThreadExtents,
200  UInt const & uiMaxBlockThreadsCount,
201  bool bRequireBlockThreadExtentsToDivideGridThreadExtents = true)
202  -> BasicWorkDiv<TDim>
203  {
204  assert(uiMaxBlockThreadsCount>0);
205  for(std::size_t i(0); i<TDim::value; ++i)
206  {
207  assert(vuiGridThreadExtents[i]>0);
208  }
209 
210  // 1. Restrict the max block thread extents with the grid thread extents.
211  // This removes dimensions not required in the given grid thread extents.
212  // This has to be done before the uiMaxBlockThreadsCount clipping to get the maximum correctly.
213  auto vuiBlockThreadExtents(Vec<TDim>::ones());
214  for(std::size_t i(0); i<TDim::value; ++i)
215  {
216  vuiBlockThreadExtents[i] = std::min(vuiMaxBlockThreadExtents[i], vuiGridThreadExtents[i]);
217  }
218 
219  // 2. If the block thread extents require more threads then available on the accelerator, clip it.
220  if(vuiBlockThreadExtents.prod() > uiMaxBlockThreadsCount)
221  {
222  // Begin in last dimension.
223  UInt uiDim(0u);
224  // Very primitive clipping. Just halve it until it fits repeatedly iterating over the dimensions.
225  while(vuiBlockThreadExtents.prod() > uiMaxBlockThreadsCount)
226  {
227  vuiBlockThreadExtents[uiDim] = std::max(static_cast<UInt>(1u), static_cast<UInt>(vuiBlockThreadExtents[uiDim] / 2u));
228  uiDim = (uiDim+1u) % TDim::value;
229  }
230  }
231 
232  if(bRequireBlockThreadExtentsToDivideGridThreadExtents)
233  {
234  // Make the block thread extents divide the grid thread extents.
235  for(std::size_t i(0); i<TDim::value; ++i)
236  {
237  vuiBlockThreadExtents[i] = detail::nextLowerOrEqualFactor(vuiBlockThreadExtents[i], vuiGridThreadExtents[i]);
238  }
239  }
240 
241  // Set the grid block extents (rounded to the next integer not less then the quotient.
242  auto vuiGridBlockExtents(Vec<TDim>::ones());
243  for(std::size_t i(0); i<TDim::value; ++i)
244  {
245  vuiGridBlockExtents[i] = static_cast<UInt>(std::ceil(static_cast<double>(vuiGridThreadExtents[0u]) / static_cast<double>(vuiBlockThreadExtents[0u])));
246  }
247 
248  return BasicWorkDiv<TDim>(vuiGridBlockExtents, vuiBlockThreadExtents);
249  }
250  }

297

311  }
