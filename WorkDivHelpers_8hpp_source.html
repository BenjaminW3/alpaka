152  //! \param uiMaxDivisor The maximum divisor.
153  //! \param uiDividend The dividend.
154  //! \return The biggest number that satisfies the following conditions:
155  //! 1) uiDividend/ret==0
156  //! 2) ret<=uiMaxDivisor
157  //-----------------------------------------------------------------------------
158  template<
159  typename T,
160  typename = typename std::enable_if<std::is_integral<T>::value && std::is_unsigned<T>::value>::type>
161  ALPAKA_FCT_HOST auto nextLowerOrEqualFactor(
162  T const & uiMaxDivisor,
163  T const & uiDividend)
164  -> T
165  {
166  T uiDivisor(uiMaxDivisor);
167  // \TODO: This is not very efficient. Replace with a better algorithm.
168  while((uiDividend%uiDivisor) != 0)
169  {
170  --uiDivisor;
171  }
172  return uiDivisor;

175  //#############################################################################
176  //! Subdivides the given grid thread extents into blocks restricted by:
177  //! 1. The maximum block thread extents
178  //! 2. The maximum block thread count
179  //! 3. The requirement of the block thread extents to divide the grid thread extents without remainder
180  //!
181  //! \param v3uiGridThreadExtents
182  //! The full extents of threads in the grid.
183  //! \param v3uiMaxBlockThreadExtents
184  //! The maximum extents of threads in a block.
185  //! \param uiMaxBlockThreadsCount
186  //! The maximum number of threads in a block.
187  //! \param bRequireBlockThreadExtentsToDivideGridThreadExtents
188  //! If this is true, the grid thread extents will be multiples of the corresponding block thread extents.
189  //! NOTE: If v3uiGridThreadExtents is prime (or otherwise bad chosen) in a dimension, the block thread extent will be one in this dimension.
190  //#############################################################################
191  ALPAKA_FCT_HOST auto subdivideGridThreads(
192  Vec<3u> const & v3uiGridThreadExtents,
193  Vec<3u> const & v3uiMaxBlockThreadExtents,
194  UInt const & uiMaxBlockThreadsCount,
195  bool bRequireBlockThreadExtentsToDivideGridThreadExtents = true)
196  -> BasicWorkDiv
197  {
198  assert(v3uiGridThreadExtents[0u]>0);
199  assert(v3uiGridThreadExtents[1u]>0);
200  assert(v3uiGridThreadExtents[2u]>0);
201 
202  // 1. Restrict the max block thread extents with the grid thread extents.
203  // This removes dimensions not required in the given grid thread extents.
204  // This has to be done before the uiMaxBlockThreadsCount clipping to get the maximum correctly.
205  Vec<3u> v3uiBlockThreadExtents(
206  std::min(v3uiMaxBlockThreadExtents[0u], v3uiGridThreadExtents[0u]),
207  std::min(v3uiMaxBlockThreadExtents[1u], v3uiGridThreadExtents[1u]),
208  std::min(v3uiMaxBlockThreadExtents[2u], v3uiGridThreadExtents[2u]));
209 
210  // 2. If the block thread extents require more threads then available on the accelerator, clip it.
211  if(v3uiBlockThreadExtents.prod() > uiMaxBlockThreadsCount)
212  {
213  // Begin in z dimension.
214  UInt uiDim(2);
215  // Very primitive clipping. Just halve it until it fits repeatedly iterating over the dimensions.
216  while(v3uiBlockThreadExtents.prod() > uiMaxBlockThreadsCount)
217  {
218  v3uiBlockThreadExtents[uiDim] = std::max(static_cast<UInt>(1u), static_cast<UInt>(v3uiBlockThreadExtents[uiDim] / 2u));
219  uiDim = (uiDim+2) % 3;
220  }
221  }
222 
223  if(bRequireBlockThreadExtentsToDivideGridThreadExtents)
224  {
225  // Make the block thread extents divide the grid thread extents.
226  v3uiBlockThreadExtents = Vec<3u>(
227  detail::nextLowerOrEqualFactor(v3uiBlockThreadExtents[0u], v3uiGridThreadExtents[0u]),
228  detail::nextLowerOrEqualFactor(v3uiBlockThreadExtents[1u], v3uiGridThreadExtents[1u]),
229  detail::nextLowerOrEqualFactor(v3uiBlockThreadExtents[2u], v3uiGridThreadExtents[2u]));
230  }
231 
232  // Set the grid block extents (rounded to the next integer not less then the quotient.
233  Vec<3u> const v3uiGridBlockExtents(
234  static_cast<UInt>(std::ceil(static_cast<double>(v3uiGridThreadExtents[0u]) / static_cast<double>(v3uiBlockThreadExtents[0u]))),
235  static_cast<UInt>(std::ceil(static_cast<double>(v3uiGridThreadExtents[1u]) / static_cast<double>(v3uiBlockThreadExtents[1u]))),
236  static_cast<UInt>(std::ceil(static_cast<double>(v3uiGridThreadExtents[2u]) / static_cast<double>(v3uiBlockThreadExtents[2u]))));
237 
238  return BasicWorkDiv(v3uiGridBlockExtents, v3uiBlockThreadExtents);
239  }
240  }

297  || (v3uiBlockThreadExtentsMax[0] < v3uiBlockThreadExtents[0])
298  || (v3uiBlockThreadExtentsMax[1] < v3uiBlockThreadExtents[1])
299  || (v3uiBlockThreadExtentsMax[2] < v3uiBlockThreadExtents[2])
300  || (uiBlockThreadCountMax < v3uiBlockThreadExtents.prod()));
