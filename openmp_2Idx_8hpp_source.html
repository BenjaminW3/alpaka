70  //-----------------------------------------------------------------------------
71  //! \return The index of the currently executed thread.
72  //
73  // \TODO: Would it be faster to precompute the 3 dimensional index and cache it inside an array?
74  //-----------------------------------------------------------------------------
75  template<
76  typename TWorkDiv>
77  ALPAKA_FCT_ACC_NO_CUDA Vec<3u> getIdxBlockThread3d(TWorkDiv const & workDiv) const
78  {
79  // We assume that the thread id is positive.
80  auto const v1iIdxBlockThread(getIdxBlockThread1d());
81  // Get the number of threads in each dimension of the grid.
82  auto const v3uiBlockThreadExtents(workdiv::getWorkDiv<Block, Threads, dim::Dim3>(workDiv));
83  auto const v2uiBlockThreadExtents(v3uiBlockThreadExtents.template subVec<2u>());
84 
85  return mapIdx<3>(
86  v1iIdxBlockThread,
87  v2uiBlockThreadExtents);
88  }
89  //-----------------------------------------------------------------------------
90  //! \return The index of the currently executed thread.
91  //-----------------------------------------------------------------------------
92  ALPAKA_FCT_ACC_NO_CUDA Vec<1u> getIdxBlockThread1d() const
93  {
94  // We assume that the thread id is positive.
95  assert(::omp_get_thread_num() >= 0);
96  auto const uiThreadId(static_cast<Vec<1u>::Val>(::omp_get_thread_num()));
97  return Vec<1u>(uiThreadId);
98  }
