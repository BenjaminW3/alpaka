177  //-----------------------------------------------------------------------------
178  //! \return Allocates block shared memory.
179  //-----------------------------------------------------------------------------
180  template<
181  typename T,
182  UInt TuiNumElements>
183  ALPAKA_FCT_ACC_NO_CUDA auto allocBlockSharedMem() const
184  -> T *
185  {
186  static_assert(TuiNumElements > 0, "The number of elements to allocate in block shared memory must not be zero!");
187 
188  // Assure that all threads have executed the return of the last allocBlockSharedMem function (if there was one before).
189  syncBlockThreads();
190 
191  // Arbitrary decision: The thread with id 0 has to allocate the memory.
192  if(::omp_get_thread_num() == 0)
193  {
194  // \TODO: C++14 std::make_unique would be better.
195  m_vvuiSharedMem.emplace_back(
196  std::unique_ptr<uint8_t[]>(
197  reinterpret_cast<uint8_t*>(new T[TuiNumElements])));
198  }
199  syncBlockThreads();
200 
201  return reinterpret_cast<T*>(m_vvuiSharedMem.back().get());
202  }

284  //! Executes the kernel functor.
285  //-----------------------------------------------------------------------------
286  template<
287  typename TKernelFunctor,
288  typename... TArgs>
289  ALPAKA_FCT_HOST auto operator()(
290  TKernelFunctor && kernelFunctor,
291  TArgs && ... args) const
292  -> void
293  {
294  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
295 
296  Vec<3u> const v3uiGridBlockExtents(this->AccOmp2::getWorkDiv<Grid, Blocks, dim::Dim3>());
297  Vec<3u> const v3uiBlockThreadExtents(this->AccOmp2::getWorkDiv<Block, Threads, dim::Dim3>());
298 
299  auto const uiBlockSharedExternMemSizeBytes(getBlockSharedExternMemSizeBytes<typename std::decay<TKernelFunctor>::type, AccOmp2>(
300  v3uiBlockThreadExtents,
301  std::forward<TArgs>(args)...));
302 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
303  std::cout << BOOST_CURRENT_FUNCTION
304  << " BlockSharedExternMemSizeBytes: " << uiBlockSharedExternMemSizeBytes << " B"
305  << std::endl;
306 #endif
307  this->AccOmp2::m_vuiExternalSharedMem.reset(
308  new uint8_t[uiBlockSharedExternMemSizeBytes]);
309 
310  // The number of threads in this block.
311  auto const uiNumThreadsInBlock(this->AccOmp2::getWorkDiv<Block, Threads, dim::Dim1>()[0]);
312 
313  // Execute the blocks serially.
314  for(this->AccOmp2::m_v3uiGridBlockIdx[2] = 0; this->AccOmp2::m_v3uiGridBlockIdx[2]<v3uiGridBlockExtents[2]; ++this->AccOmp2::m_v3uiGridBlockIdx[2])
315  {
316  for(this->AccOmp2::m_v3uiGridBlockIdx[1] = 0; this->AccOmp2::m_v3uiGridBlockIdx[1]<v3uiGridBlockExtents[1]; ++this->AccOmp2::m_v3uiGridBlockIdx[1])
317  {
318  for(this->AccOmp2::m_v3uiGridBlockIdx[0] = 0; this->AccOmp2::m_v3uiGridBlockIdx[0]<v3uiGridBlockExtents[0]; ++this->AccOmp2::m_v3uiGridBlockIdx[0])
319  {
320  // Execute the threads in parallel.
321 
322  // Force the environment to use the given number of threads.
323  ::omp_set_dynamic(0);
324 
325  // Parallel execution of the threads in a block is required because when syncBlockThreads is called all of them have to be done with their work up to this line.
326  // So we have to spawn one OS thread per thread in a block.
327  // 'omp for' is not useful because it is meant for cases where multiple iterations are executed by one thread but in our case a 1:1 mapping is required.
328  // Therefore we use 'omp parallel' with the specified number of threads in a block.
329  //
330  // \TODO: Does this hinder executing multiple threads in parallel because their block sizes/omp thread numbers are interfering? Is this num_threads global? Is this a real use case?
331  #pragma omp parallel num_threads(static_cast<int>(uiNumThreadsInBlock))
332  {
333 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
334  if((::omp_get_thread_num() == 0) && (this->AccOmp2::m_v3uiGridBlockIdx[2] == 0) && (this->AccOmp2::m_v3uiGridBlockIdx[1] == 0) && (this->AccOmp2::m_v3uiGridBlockIdx[0] == 0))
335  {
336  assert(::omp_get_num_threads()>=0);
337  auto const uiNumThreads(static_cast<decltype(uiNumThreadsInBlock)>(::omp_get_num_threads()));
338  std::cout << "omp_get_num_threads: " << uiNumThreads << std::endl;
339  if(uiNumThreads != uiNumThreadsInBlock)
340  {
341  throw std::runtime_error("The OpenMP2 runtime did not use the number of threads that had been required!");
342  }
343  }
344 #endif
345  std::forward<TKernelFunctor>(kernelFunctor)(
346  (*static_cast<AccOmp2 const *>(this)),
347  std::forward<TArgs>(args)...);
348 
349  // Wait for all threads to finish before deleting the shared memory.
350  this->AccOmp2::syncBlockThreads();
351  }
352 
353  // After a block has been processed, the shared memory can be deleted.
354  this->AccOmp2::m_vvuiSharedMem.clear();
355  }
356  }
357  }
358  // After all blocks have been processed, the external shared memory can be deleted.
359  this->AccOmp2::m_vuiExternalSharedMem.reset();
360  }
361  };
362  }
