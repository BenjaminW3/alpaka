187  //-----------------------------------------------------------------------------
188  //! \return The pointer to the externally allocated block shared memory.
189  //-----------------------------------------------------------------------------
190  template<
191  typename T>
192  ALPAKA_FCT_ACC_CUDA_ONLY auto getBlockSharedExternMem() const
193  -> T *
194  {
195  // Because unaligned access to variables is not allowed in device code,
196  // we have to use the widest possible type to have all types aligned correctly.
197  // See: http://docs.nvidia.com/cuda/cuda-c-programming-guide/index.html#shared
198  // http://docs.nvidia.com/cuda/cuda-c-programming-guide/index.html#vector-types
199  extern __shared__ float4 shMem[];
200  return reinterpret_cast<T *>(shMem);
201  }
