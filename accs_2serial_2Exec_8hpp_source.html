113  //-----------------------------------------------------------------------------
114  //! Executes the kernel functor.
115  //-----------------------------------------------------------------------------
116  template<
117  typename TKernelFunctor,
118  typename... TArgs>
119  ALPAKA_FCT_HOST auto operator()(
120  TKernelFunctor && kernelFunctor,
121  TArgs && ... args) const
122  -> void
123  {
124  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
125 
126  auto const vuiGridBlockExtents(this->AccCpuSerial<TDim>::template getWorkDiv<Grid, Blocks>());
127  auto const vuiBlockThreadExtents(this->AccCpuSerial<TDim>::template getWorkDiv<Block, Threads>());
128 
129  auto const uiBlockSharedExternMemSizeBytes(
130  kernel::getBlockSharedExternMemSizeBytes<
131  typename std::decay<TKernelFunctor>::type,
132  AccCpuSerial<TDim>>(
133  vuiBlockThreadExtents,
134  std::forward<TArgs>(args)...));
135  #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
136  std::cout << BOOST_CURRENT_FUNCTION
137  << " BlockSharedExternMemSizeBytes: " << uiBlockSharedExternMemSizeBytes << " B"
138  << std::endl;
139  #endif
140  if(uiBlockSharedExternMemSizeBytes > 0)
141  {
142  this->AccCpuSerial<TDim>::m_vuiExternalSharedMem.reset(
143  new uint8_t[uiBlockSharedExternMemSizeBytes]);
144  }
145 
146  // Execute the blocks serially.
147  ndLoop(
148  vuiGridBlockExtents,
149  [&](Vec<TDim> const & vuiBlockThreadIdx)
150  {
151  this->AccCpuSerial<TDim>::m_vuiGridBlockIdx = vuiBlockThreadIdx;
152  assert(vuiBlockThreadExtents.prod() == 1u);
153 
154  // There is only ever one thread in a block in the serial accelerator.
155  std::forward<TKernelFunctor>(kernelFunctor)(
156  (*static_cast<AccCpuSerial<TDim> const *>(this)),
157  std::forward<TArgs>(args)...);
158 
159  // After a block has been processed, the shared memory has to be deleted.
160  this->AccCpuSerial<TDim>::m_vvuiSharedMem.clear();
161  });
162 
163  // After all blocks have been processed, the external shared memory has to be deleted.
164  this->AccCpuSerial<TDim>::m_vuiExternalSharedMem.reset();
165  }
166 
167  public:
168  devs::cpu::StreamCpu m_Stream;
169  };
170  }
