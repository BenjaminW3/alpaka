273  //-----------------------------------------------------------------------------
274  //! Executes the kernel functor.
275  //-----------------------------------------------------------------------------
276  template<
277  typename TKernelFunctor,
278  typename... TArgs>
279  ALPAKA_FCT_HOST auto operator()(
280  TKernelFunctor && kernelFunctor,
281  TArgs && ... args) const
282  -> void
283  {
284  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
285 
286  Vec<3u> const v3uiGridBlockExtents(this->AccSerial::getWorkDiv<Grid, Blocks, dim::Dim3>());
287  Vec<3u> const v3uiBlockThreadExtents(this->AccSerial::getWorkDiv<Block, Threads, dim::Dim3>());
288 
289  auto const uiBlockSharedExternMemSizeBytes(getBlockSharedExternMemSizeBytes<typename std::decay<TKernelFunctor>::type, AccSerial>(
290  v3uiBlockThreadExtents,
291  std::forward<TArgs>(args)...));
292  #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
293  std::cout << BOOST_CURRENT_FUNCTION
294  << " BlockSharedExternMemSizeBytes: " << uiBlockSharedExternMemSizeBytes << " B"
295  << std::endl;
296  #endif
297  this->AccSerial::m_vuiExternalSharedMem.reset(
298  new uint8_t[uiBlockSharedExternMemSizeBytes]);
299 
300  // Execute the blocks serially.
301  for(this->AccSerial::m_v3uiGridBlockIdx[2] = 0; this->AccSerial::m_v3uiGridBlockIdx[2]<v3uiGridBlockExtents[2]; ++this->AccSerial::m_v3uiGridBlockIdx[2])
302  {
303  for(this->AccSerial::m_v3uiGridBlockIdx[1] = 0; this->AccSerial::m_v3uiGridBlockIdx[1]<v3uiGridBlockExtents[1]; ++this->AccSerial::m_v3uiGridBlockIdx[1])
304  {
305  for(this->AccSerial::m_v3uiGridBlockIdx[0] = 0; this->AccSerial::m_v3uiGridBlockIdx[0]<v3uiGridBlockExtents[0]; ++this->AccSerial::m_v3uiGridBlockIdx[0])
306  {
307  assert(v3uiBlockThreadExtents[0] == 1);
308  assert(v3uiBlockThreadExtents[1] == 1);
309  assert(v3uiBlockThreadExtents[2] == 1);
310 
311  // There is only ever one thread in a block in the serial accelerator.
312  std::forward<TKernelFunctor>(kernelFunctor)(
313  (*static_cast<AccSerial const *>(this)),
314  std::forward<TArgs>(args)...);
315 
316  // After a block has been processed, the shared memory can be deleted.
317  this->AccSerial::m_vvuiSharedMem.clear();
318  }
319  }
320  }
321  // After all blocks have been processed, the external shared memory can be deleted.
322  this->AccSerial::m_vuiExternalSharedMem.reset();
323  }
324  };
325  }
326  }
