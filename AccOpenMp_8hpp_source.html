173  template<
174  typename T,
175  UInt TuiNumElements>
176  ALPAKA_FCT_ACC_NO_CUDA T * allocBlockSharedMem() const
177  {
178  static_assert(TuiNumElements > 0, "The number of elements to allocate in block shared memory must not be zero!");
179 
180  // Assure that all threads have executed the return of the last allocBlockSharedMem function (if there was one before).
181  syncBlockThreads();
182 
183  // Arbitrary decision: The thread with id 0 has to allocate the memory.
184  if(::omp_get_thread_num() == 0)
185  {
186  // \TODO: C++14 std::make_unique would be better.
187  m_vvuiSharedMem.emplace_back(
188  std::unique_ptr<uint8_t[]>(
189  reinterpret_cast<uint8_t*>(new T[TuiNumElements])));
190  }
191  syncBlockThreads();
192 
193  return reinterpret_cast<T*>(m_vvuiSharedMem.back().get());
194  }

282  //-----------------------------------------------------------------------------
283  //! Executes the accelerated kernel.
284  //-----------------------------------------------------------------------------
285  template<
286  typename... TArgs>
287  ALPAKA_FCT_HOST void operator()(
288  TArgs && ... args) const
289  {
290  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
291 
292  Vec<3u> const v3uiGridBlockExtents(this->AccOpenMp::getWorkDiv<Grid, Blocks, dim::Dim3>());
293  Vec<3u> const v3uiBlockThreadExtents(this->AccOpenMp::getWorkDiv<Block, Threads, dim::Dim3>());
294 
295  auto const uiBlockSharedExternMemSizeBytes(BlockSharedExternMemSizeBytes<TAcceleratedKernel>::getBlockSharedExternMemSizeBytes(v3uiBlockThreadExtents, std::forward<TArgs>(args)...));
296  this->AccOpenMp::m_vuiExternalSharedMem.reset(
297  new uint8_t[uiBlockSharedExternMemSizeBytes]);
298 
299  // The number of threads in this block.
300  auto const uiNumThreadsInBlock(this->AccOpenMp::getWorkDiv<Block, Threads, dim::Dim1>()[0]);
301 
302  // Execute the blocks serially.
303  for(std::uint32_t bz(0); bz<v3uiGridBlockExtents[2]; ++bz)
304  {
305  this->AccOpenMp::m_v3uiGridBlockIdx[2] = bz;
306  for(std::uint32_t by(0); by<v3uiGridBlockExtents[1]; ++by)
307  {
308  this->AccOpenMp::m_v3uiGridBlockIdx[1] = by;
309  for(std::uint32_t bx(0); bx<v3uiGridBlockExtents[0]; ++bx)
310  {
311  this->AccOpenMp::m_v3uiGridBlockIdx[0] = bx;
312 
313  // Execute the threads in parallel threads.
314 
315  // Force the environment to use the given number of threads.
316  ::omp_set_dynamic(0);
317 
318  // Parallel execution of the threads in a block is required because when syncBlockThreads is called all of them have to be done with their work up to this line.
319  // So we have to spawn one OS thread per thread in a block.
320  // 'omp for' is not useful because it is meant for cases where multiple iterations are executed by one thread but in our case a 1:1 mapping is required.
321  // Therefore we use 'omp parallel' with the specified number of threads in a block.
322  //
323  // \TODO: Does this hinder executing multiple threads in parallel because their block sizes/omp thread numbers are interfering? Is this num_threads global? Is this a real use case?
324  #pragma omp parallel num_threads(static_cast<int>(uiNumThreadsInBlock))
325  {
326 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
327  if((::omp_get_thread_num() == 0) && (bz == 0) && (by == 0) && (bx == 0))
328  {
329  assert(::omp_get_num_threads()>=0);
330  auto const uiNumThreads(static_cast<decltype(uiNumThreadsInBlock)>(::omp_get_num_threads()));
331  std::cout << "omp_get_num_threads: " << uiNumThreads << std::endl;
332  if(uiNumThreads != uiNumThreadsInBlock)
333  {
334  throw std::runtime_error("The OpenMP runtime did not use the number of threads that had been required!");
335  }
336  }
337 #endif
338  this->TAcceleratedKernel::operator()(
339  (*static_cast<IAcc<AccOpenMp> const *>(this)),
340  std::forward<TArgs>(args)...);
341 
342  // Wait for all threads to finish before deleting the shared memory.
343  this->AccOpenMp::syncBlockThreads();
344  }
345 
346  // After a block has been processed, the shared memory can be deleted.
347  this->AccOpenMp::m_vvuiSharedMem.clear();
348  }
349  }
350  }
351  // After all blocks have been processed, the external shared memory can be deleted.
352  this->AccOpenMp::m_vuiExternalSharedMem.reset();
353  }
354  };
