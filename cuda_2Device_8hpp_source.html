148  //! Sets the device to use with this accelerator.
149  //-----------------------------------------------------------------------------
150  ALPAKA_FCT_HOST static void setCurrentDev(
151  cuda::detail::DevCuda const & device)
152  {
153  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
154 
155  std::size_t uiNumDevices(getDevCount());
156  if(uiNumDevices < 1)
157  {
158  throw std::runtime_error("No CUDA capable devices detected!");
159  }
160  else if(uiNumDevices < device.m_iDevice)
161  {
162  std::stringstream ssErr;
163  ssErr << "No CUDA device " << device.m_iDevice << " available, only " << uiNumDevices << " devices found!";
164  throw std::runtime_error(ssErr.str());
165  }
166 
167  cudaDeviceProp devProp;
168  ALPAKA_CUDA_CHECK(cudaGetDeviceProperties(&devProp, device.m_iDevice));
169  // Default compute mode (Multiple threads can use cudaSetDevice() with this device)
170  if(devProp.computeMode == cudaComputeModeDefault)
171  {
172  ALPAKA_CUDA_CHECK(cudaSetDevice(device.m_iDevice));
173  std::cout << "Set device to " << device.m_iDevice << ": " << std::endl;
174 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
175  printDeviceProperties(devProp);
176 #else
177  std::cout << devProp.name << std::endl;
178 #endif
179  }
180  // Compute-exclusive-thread mode (Only one thread in one process will be able to use cudaSetDevice() with this device)
181  else if(devProp.computeMode == cudaComputeModeExclusive)
182  {
183  std::cout << "Requested device is in computeMode cudaComputeModeExclusive.";
184  // \TODO: Are we allowed to use the device in compute mode cudaComputeModeExclusive?
185  }
186  // Compute-prohibited mode (No threads can use cudaSetDevice() with this device)
187  else if(devProp.computeMode == cudaComputeModeProhibited)
188  {
189  std::cout << "Requested device is in computeMode cudaComputeModeProhibited. It can not be selected!";
190  }
191  // Compute-exclusive-process mode (Many threads in one process will be able to use cudaSetDevice() with this device)
192  else if(devProp.computeMode == cudaComputeModeExclusiveProcess)
193  {
194  std::cerr << "Requested device is in computeMode cudaComputeModeExclusiveProcess.";
195  // \TODO: Are we allowed to use the device in compute mode cudaComputeModeExclusiveProcess?
196  }
197  else
198  {
199  std::cerr << "unknown computeMode!";
200  }
201 
202  // Instruct CUDA to actively spin when waiting for results from the device.
203  // This can decrease latency when waiting for the device, but may lower the performance of CPU threads if they are performing work in parallel with the CUDA thread.
204  ALPAKA_CUDA_CHECK(cudaSetDeviceFlags(cudaDeviceScheduleSpin));
205  }
