178  -> void
179  {

182 
183  syncBlockThreads(itFind);
184  }
185  private:
186  //-----------------------------------------------------------------------------
187  //! Syncs all threads in the current block.
188  //-----------------------------------------------------------------------------
189  ALPAKA_FCT_ACC_NO_CUDA auto syncBlockThreads(
190  std::map<std::thread::id, UInt>::iterator const & itFind) const
191  -> void
192  {
193  assert(itFind != m_mThreadsToBarrier.end());
194 
195  auto & uiBarrierIdx(itFind->second);
196  std::size_t const uiModBarrierIdx(uiBarrierIdx % 2);
197 
198  auto & bar(m_abarSyncThreads[uiModBarrierIdx]);
199 
200  // (Re)initialize a barrier if this is the first thread to reach it.
201  if(bar.getNumThreadsToWaitFor() == 0)
202  {
203  std::lock_guard<std::mutex> lock(m_mtxBarrier);
204  if(bar.getNumThreadsToWaitFor() == 0)
205  {
206  bar.reset(m_uiNumThreadsPerBlock);
207  }
208  }
209 
210  // Wait for the barrier.
211  bar.wait();
212  ++uiBarrierIdx;
213  }

365  //-----------------------------------------------------------------------------
366  //! Executes the kernel functor.
367  //-----------------------------------------------------------------------------
368  template<
369  typename TKernelFunctor,
370  typename... TArgs>
371  ALPAKA_FCT_HOST auto operator()(
372  TKernelFunctor && kernelFunctor,
373  TArgs && ... args) const
374  -> void
375  {
376  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
377 
378  Vec<3u> const v3uiGridBlockExtents(this->AccThreads::getWorkDiv<Grid, Blocks, dim::Dim3>());
379  Vec<3u> const v3uiBlockThreadExtents(this->AccThreads::getWorkDiv<Block, Threads, dim::Dim3>());
380 
381  auto const uiBlockSharedExternMemSizeBytes(getBlockSharedExternMemSizeBytes<typename std::decay<TKernelFunctor>::type, AccThreads>(
382  v3uiBlockThreadExtents,
383  std::forward<TArgs>(args)...));
384  #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
385  std::cout << BOOST_CURRENT_FUNCTION
386  << " BlockSharedExternMemSizeBytes: " << uiBlockSharedExternMemSizeBytes << " B"
387  << std::endl;
388  #endif
389  this->AccThreads::m_vuiExternalSharedMem.reset(
390  new uint8_t[uiBlockSharedExternMemSizeBytes]);
391 
392  auto const uiNumThreadsInBlock(this->AccThreads::getWorkDiv<Block, Threads, dim::Dim1>());
393  // When using the thread pool the threads are yielding because this is faster.
394  // Using condition variables and going to sleep is very costly for real threads.
395  // Especially when the time to wait is really short (syncBlockThreads) yielding is much faster.
396  using ThreadPool = alpaka::detail::ConcurrentExecPool<
397  std::thread, // The concurrent execution type.
398  std::promise, // The promise type.
399  ThreadPoolCurrentException, // The type returning the current exception.
400  ThreadPoolYield>; // The type yielding the current concurrent execution.
401  ThreadPool pool(uiNumThreadsInBlock[0], uiNumThreadsInBlock[0]);
402 
403  // Execute the blocks serially.
404  for(this->AccThreads::m_v3uiGridBlockIdx[2] = 0; this->AccThreads::m_v3uiGridBlockIdx[2]<v3uiGridBlockExtents[2]; ++this->AccThreads::m_v3uiGridBlockIdx[2])
405  {
406  for(this->AccThreads::m_v3uiGridBlockIdx[1] = 0; this->AccThreads::m_v3uiGridBlockIdx[1]<v3uiGridBlockExtents[1]; ++this->AccThreads::m_v3uiGridBlockIdx[1])
407  {
408  for(this->AccThreads::m_v3uiGridBlockIdx[0] = 0; this->AccThreads::m_v3uiGridBlockIdx[0]<v3uiGridBlockExtents[0]; ++this->AccThreads::m_v3uiGridBlockIdx[0])
409  {
410  // Execute the threads in parallel.
411  Vec<3u> v3uiBlockThreadIdx(0u);
412  for(v3uiBlockThreadIdx[2] = 0; v3uiBlockThreadIdx[2]<v3uiBlockThreadExtents[2]; ++v3uiBlockThreadIdx[2])
413  {
414  for(v3uiBlockThreadIdx[1] = 0; v3uiBlockThreadIdx[1]<v3uiBlockThreadExtents[1]; ++v3uiBlockThreadIdx[1])
415  {
416  for(v3uiBlockThreadIdx[0] = 0; v3uiBlockThreadIdx[0]<v3uiBlockThreadExtents[0]; ++v3uiBlockThreadIdx[0])
417  {
418  // The v3uiBlockThreadIdx is required to be copied in from the environment because if the thread is immediately suspended the variable is already changed for the next iteration/thread.
419  auto threadKernelFct =
420  [&, v3uiBlockThreadIdx]()
421  {
422  threadKernel(
423  v3uiBlockThreadIdx,
424  std::forward<TKernelFunctor>(kernelFunctor),
425  std::forward<TArgs>(args)...);
426  };
427  m_vFuturesInBlock.emplace_back(
428  pool.enqueueTask(
429  threadKernelFct));
430  }
431  }
432  }
433 
434  // Join all the threads.
435  std::for_each(m_vFuturesInBlock.begin(), m_vFuturesInBlock.end(),
436  [](std::future<void> & t)
437  {
438  t.wait();
439  }
440  );
441  // Clean up.
442  m_vFuturesInBlock.clear();
443 
444  this->AccThreads::m_mThreadsToIndices.clear();
445  this->AccThreads::m_mThreadsToBarrier.clear();
446 
447  // After a block has been processed, the shared memory can be deleted.
448  this->AccThreads::m_vvuiSharedMem.clear();
449  }
450  }
451  }
452  // After all blocks have been processed, the external shared memory can be deleted.
453  this->AccThreads::m_vuiExternalSharedMem.reset();
454  }
455  private:
456  //-----------------------------------------------------------------------------
457  //! The thread entry point.
458  //-----------------------------------------------------------------------------
459  template<
460  typename TKernelFunctor,
461  typename... TArgs>
462  ALPAKA_FCT_HOST auto threadKernel(
463  Vec<3u> const & v3uiBlockThreadIdx,
464  TKernelFunctor && kernelFunctor,
465  TArgs && ... args) const
466  -> void
467  {
468  // We have to store the thread data before the kernel is calling any of the methods of this class depending on them.
469  auto const idThread(std::this_thread::get_id());
470 
471  // Set the master thread id.
472  if(v3uiBlockThreadIdx[0] == 0 && v3uiBlockThreadIdx[1] == 0 && v3uiBlockThreadIdx[2] == 0)
473  {
474  this->AccThreads::m_idMasterThread = idThread;
475  }
476 
477  // We can not use the default syncBlockThreads here because it searches inside m_mThreadsToBarrier for the thread id.
478  // Concurrently searching while others use emplace is unsafe!
479  std::map<std::thread::id, UInt>::iterator itThreadToBarrier;
480 
481  {
482  // The insertion of elements has to be done one thread at a time.
483  std::lock_guard<std::mutex> lock(m_mtxMapInsert);
484 
485  // Save the thread id, and index.
486  this->AccThreads::m_mThreadsToIndices.emplace(idThread, v3uiBlockThreadIdx);
487  itThreadToBarrier = this->AccThreads::m_mThreadsToBarrier.emplace(idThread, 0).first;
488  }
489 
490  // Sync all fibers so that the maps with fiber id's are complete and not changed after here.
491  this->AccThreads::syncBlockThreads(itThreadToBarrier);
492 
493  // Execute the kernel itself.
494  std::forward<TKernelFunctor>(kernelFunctor)(
495  (*static_cast<AccThreads const *>(this)),
496  std::forward<TArgs>(args)...);
497 
498  // We have to sync all threads here because if a thread would finish before all threads have been started, the new thread could get a recycled (then duplicate) thread id!
499  this->AccThreads::syncBlockThreads(itThreadToBarrier);
500  }

504 
505  std::mutex mutable m_mtxMapInsert;
506  };
507  }
