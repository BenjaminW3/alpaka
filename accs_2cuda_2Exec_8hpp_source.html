122  //-----------------------------------------------------------------------------
123  //! Executes the kernel functor.
124  //-----------------------------------------------------------------------------
125  template<
126  typename TKernelFunctor,
127  typename... TArgs>
128  ALPAKA_FCT_HOST auto operator()(
129  // \NOTE: No universal reference (&&) or const reference (const &) is allowed as the parameter type because the kernel launch language extension expects the arguments by value.
130  // This forces the type of a float argument given with std::forward to this function to be of type float instead of e.g. "float const & __ptr64" (MSVC).
131  // If not given by value, the kernel launch code does not copy the value but the pointer to the value location.
132  TKernelFunctor kernelFunctor,
133  TArgs ... args) const
134  -> void
135  {
136 #if (!__GLIBCXX__) // libstdc++ even for gcc-4.9 does not support std::is_trivially_copyable.
137  static_assert(std::is_trivially_copyable<TKernelFunctor>::value, "The given kernel functor has to fulfill is_trivially_copyable!");
138 #endif
139  // TODO: Check that (sizeof(TKernelFunctor) * m_3uiBlockThreadExtents.prod()) < available memory size
140 
141  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
142 
143 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
144  //std::size_t uiPrintfFifoSize;
145  //cudaDeviceGetLimit(&uiPrintfFifoSize, cudaLimitPrintfFifoSize);
146  //std::cout << "uiPrintfFifoSize: " << uiPrintfFifoSize << std::endl;
147  //cudaDeviceSetLimit(cudaLimitPrintfFifoSize, uiPrintfFifoSize*10);
148  //cudaDeviceGetLimit(&uiPrintfFifoSize, cudaLimitPrintfFifoSize);
149  //std::cout << "uiPrintfFifoSize: " << uiPrintfFifoSize << std::endl;
150 #endif
151  dim3 gridDim(1u, 1u, 1u);
152  dim3 blockDim(1u, 1u, 1u);
153  for(std::size_t i(0u); i<3u; ++i)
154  {
155  reinterpret_cast<unsigned int *>(&gridDim)[i] = m_vuiGridBlockExtents[TDim::value-1u-i];
156  reinterpret_cast<unsigned int *>(&blockDim)[i] = m_vuiBlockThreadExtents[TDim::value-1u-i];
157  }
158  for(std::size_t i(3u); i<TDim::value; ++i)
159  {
160  assert(m_vuiGridBlockExtents[TDim::value-1u-i] == 1);
161  }
162 
163 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
164  std::cout << "gridDim: " << gridDim.z << " " << gridDim.y << " " << gridDim.x << std::endl;
165  std::cout << "blockDim: " << blockDim.z << " " << blockDim.y << " " << blockDim.x << std::endl;
166 #endif
167 
168  // Get the size of the block shared extern memory.
169  auto const uiBlockSharedExternMemSizeBytes(
170  kernel::getBlockSharedExternMemSizeBytes<
171  typename std::decay<TKernelFunctor>::type,
172  AccGpuCuda<TDim>>(
173  m_vuiBlockThreadExtents,
174  std::forward<TArgs>(args)...));
175 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
176  // Log the block shared memory size.
177  std::cout << BOOST_CURRENT_FUNCTION
178  << " BlockSharedExternMemSizeBytes: " << uiBlockSharedExternMemSizeBytes << " B"
179  << std::endl;
180 #endif
181 
182 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
183  // Log the function attributes.
184  cudaFuncAttributes funcAttrs;
185  cudaFuncGetAttributes(&funcAttrs, cudaKernel<TDim, TKernelFunctor, TArgs...>);
186  std::cout << BOOST_CURRENT_FUNCTION
187  << "binaryVersion: " << funcAttrs.binaryVersion
188  << "constSizeBytes: " << funcAttrs.constSizeBytes << " B"
189  << "localSizeBytes: " << funcAttrs.localSizeBytes << " B"
190  << "maxThreadsPerBlock: " << funcAttrs.maxThreadsPerBlock
191  << "numRegs: " << funcAttrs.numRegs
192  << "ptxVersion: " << funcAttrs.ptxVersion
193  << "sharedSizeBytes: " << funcAttrs.sharedSizeBytes << " B"
194  << std::endl;
195 #endif
196 
197  // Set the current device.
198  ALPAKA_CUDA_RT_CHECK(cudaSetDevice(
199  m_Stream.m_Dev.m_iDevice));
200  // Enqueue the kernel execution.
201  cudaKernel<TDim, TKernelFunctor, TArgs...><<<
202  gridDim,
203  blockDim,
204  uiBlockSharedExternMemSizeBytes,
205  *m_Stream.m_spCudaStream.get()>>>(
206  kernelFunctor,
207  args...);
208 
209 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
210  // Wait for the kernel execution to finish but do not check error return of this call.
211  // Do not use the alpaka::wait method because it checks the error itself but we want to give a custom error message.
212  cudaStreamSynchronize(*m_Stream.m_spCudaStream.get());
213  //cudaDeviceSynchronize();
214  cudaError_t const error(cudaGetLastError());
215  if(error != cudaSuccess)
216  {
217  std::string const sError("The execution of kernel '" + std::string(typeid(TKernelFunctor).name()) + " failed with error: '" + std::string(cudaGetErrorString(error)) + "'");
218  std::cerr << sError << std::endl;
219  ALPAKA_DEBUG_BREAK;
220  throw std::runtime_error(sError);
221  }
222 #endif
223  }
