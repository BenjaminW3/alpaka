156  //! Builds a kernel executor.
157  //!
158  //! Requirements for type TKernel:
159  //! The kernel type has to be inherited from 'alpaka::IAcc<boost::mpl::_1>'directly.
160  //! All template parameters have to be types. No value parameters are allowed. Use boost::mpl::int_ or similar to use values.
161  //#############################################################################

164  typename TKernel,
165  typename... TKernelConstrArgs>
166  auto createKernelExecutor(
167  TKernelConstrArgs && ... args)
168  -> typename std::result_of<detail::KernelExecCreator<TAcc, TKernel, TKernelConstrArgs...>(TKernelConstrArgs...)>::type
169  {
170 #if (!__GLIBCXX__) // libstdc++ even for gcc-4.9 does not support std::is_trivially_copyable.

172 #endif
173 
174  // Use the specialized KernelExecCreator for the given accelerator.
175  return detail::KernelExecCreator<TAcc, TKernel, TKernelConstrArgs...>()(std::forward<TKernelConstrArgs>(args)...);
