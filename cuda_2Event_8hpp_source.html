48  ALPAKA_FCT_HOST EventCuda(bool bBusyWait = true) :
49  m_spCudaEvent(new cudaEvent_t, &EventCuda::destroyEvent)
50  {
51  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
52 
53  // Creates an event object with the specified flags. Valid flags include:
54  // cudaEventDefault: Default event creation flag.
55  // cudaEventBlockingSync : Specifies that event should use blocking synchronization.A host thread that uses cudaEventSynchronize() to wait on an event created with this flag will block until the event actually completes.
56  // cudaEventDisableTiming : Specifies that the created event does not need to record timing data.Events created with this flag specified and the cudaEventBlockingSync flag not specified will provide the best performance when used with cudaStreamWaitEvent() and cudaEventQuery().
57  ALPAKA_CUDA_CHECK(cudaEventCreateWithFlags(
58  m_spCudaEvent.get(),
59  (bBusyWait ? cudaEventDefault : cudaEventBlockingSync) | cudaEventDisableTiming));
60  }
