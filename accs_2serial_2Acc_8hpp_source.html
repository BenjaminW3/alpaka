165  //-----------------------------------------------------------------------------
166  //! \return Allocates block shared memory.
167  //-----------------------------------------------------------------------------
168  template<
169  typename T,
170  UInt TuiNumElements>
171  ALPAKA_FCT_ACC_NO_CUDA auto allocBlockSharedMem() const
172  -> T *
173  {
174  static_assert(TuiNumElements > 0, "The number of elements to allocate in block shared memory must not be zero!");
175 
176  // \TODO: C++14 std::make_unique would be better.
177  m_vvuiSharedMem.emplace_back(
178  std::unique_ptr<uint8_t[]>(
179  reinterpret_cast<uint8_t*>(new T[TuiNumElements])));
180  return reinterpret_cast<T*>(m_vvuiSharedMem.back().get());
181  }
182 
183  //-----------------------------------------------------------------------------
184  //! \return The pointer to the externally allocated block shared memory.
185  //-----------------------------------------------------------------------------
186  template<
187  typename T>
188  ALPAKA_FCT_ACC_NO_CUDA auto getBlockSharedExternMem() const
189  -> T *
190  {
191  return reinterpret_cast<T*>(m_vuiExternalSharedMem.get());
192  }
193
