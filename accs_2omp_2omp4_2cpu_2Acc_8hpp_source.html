172  //-----------------------------------------------------------------------------
173  //! \return Allocates block shared memory.
174  //-----------------------------------------------------------------------------
175  template<
176  typename T,
177  UInt TuiNumElements>
178  ALPAKA_FCT_ACC_NO_CUDA auto allocBlockSharedMem() const
179  -> T *
180  {
181  static_assert(TuiNumElements > 0, "The number of elements to allocate in block shared memory must not be zero!");
182 
183  // Assure that all threads have executed the return of the last allocBlockSharedMem function (if there was one before).
184  syncBlockThreads();
185 
186  // Arbitrary decision: The thread with id 0 has to allocate the memory.
187  if(::omp_get_thread_num() == 0)
188  {
189  // \TODO: C++14 std::make_unique would be better.
190  m_vvuiSharedMem.emplace_back(
191  std::unique_ptr<uint8_t[]>(
192  reinterpret_cast<uint8_t*>(new T[TuiNumElements])));
193  }
194  syncBlockThreads();
195 
196  return reinterpret_cast<T*>(m_vvuiSharedMem.back().get());
197  }
198
